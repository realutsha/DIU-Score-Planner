<script>
    function animateValue(obj, start, end, duration) {
        let startTimestamp = null;
        const step = (timestamp) => {
            if (!startTimestamp) startTimestamp = timestamp;
            const progress = Math.min((timestamp - startTimestamp) / duration, 1);
            obj.innerHTML = (progress * (end - start) + start).toFixed(2).replace(/\.00$/, '');
            if (progress < 1) {
                window.requestAnimationFrame(step);
            }
        };
        window.requestAnimationFrame(step);
    }

    function calculateTarget() {
        // 1. Fetch Values
        let q1 = parseFloat(document.getElementById('q1').value);
        let q2 = parseFloat(document.getElementById('q2').value);
        let q3 = parseFloat(document.getElementById('q3').value);
        let mid = parseFloat(document.getElementById('mid').value);
        let att = parseFloat(document.getElementById('att').value);
        let pres = parseFloat(document.getElementById('pres').value);
        let assign = parseFloat(document.getElementById('assign').value);

        const scoreDisplay = document.getElementById('scoreDisplay');
        const progressRing = document.getElementById('progressRing');
        const alertBox = document.getElementById('alertBox');
        
        // Reset UI states
        alertBox.classList.remove('show', 'alert-success', 'alert-danger');

        // ==========================================
        // NEW: STRICT INPUT VALIDATION
        // ==========================================
        let errors = [];
        
        if (q1 > 15 || q2 > 15 || q3 > 15) errors.push("Quiz marks cannot exceed 15.");
        if (q1 < 0 || q2 < 0 || q3 < 0) errors.push("Quiz marks cannot be negative.");
        
        if (mid > 25) errors.push("Midterm marks cannot exceed 25.");
        if (mid < 0) errors.push("Midterm marks cannot be negative.");
        
        if (att > 7) errors.push("Attendance cannot exceed 7.");
        if (att < 0) errors.push("Attendance cannot be negative.");
        
        if (pres > 8) errors.push("Presentation cannot exceed 8.");
        if (pres < 0) errors.push("Presentation cannot be negative.");
        
        if (assign > 5) errors.push("Assignment cannot exceed 5.");
        if (assign < 0) errors.push("Assignment cannot be negative.");

        // If any errors exist, show them and STOP the calculation
        if (errors.length > 0) {
            scoreDisplay.innerHTML = "Err";
            progressRing.style.strokeDashoffset = 628; // Empty the ring
            
            alertBox.innerHTML = `⚠️ <strong>Invalid Input Detected:</strong><br><br>${errors.join('<br>')}`;
            alertBox.classList.add('alert-danger');
            
            setTimeout(() => alertBox.classList.add('show'), 100);
            return; // Halt function execution
        }
        // ==========================================

        // Normalize NaNs to 0 for valid empty fields
        mid = mid || 0;
        att = att || 0;
        pres = pres || 0;
        assign = assign || 0;

        let quizzesTaken = 0;
        let quizSum = 0;

        if(!isNaN(q1)) { quizzesTaken++; quizSum += q1; }
        if(!isNaN(q2)) { quizzesTaken++; quizSum += q2; }
        if(!isNaN(q3)) { quizzesTaken++; quizSum += q3; }

        let quizAverage = quizzesTaken > 0 ? quizSum / 3 : 0;
        
        let currentTotal = mid + quizAverage + att + pres + assign;
        let neededMarks = 80 - currentTotal;

        // Animate Number & SVG Ring
        animateValue(scoreDisplay, 0, currentTotal, 1200);

        const radius = progressRing.r.baseVal.value; 
        const circumference = radius * 2 * Math.PI;  
        progressRing.style.strokeDasharray = `${circumference} ${circumference}`;
        
        let visualTotal = currentTotal > 80 ? 80 : currentTotal;
        const offset = circumference - (visualTotal / 80) * circumference;
        
        setTimeout(() => {
            progressRing.style.strokeDashoffset = offset;
        }, 50);

        // Generate Final Suggestion Box Text
        let alertHTML = '';
        
        if (currentTotal === 0 && quizzesTaken === 0) {
            alertHTML = `Enter your current scores on the left, and hit <strong>Calculate</strong> to map out your path to an 80+ grade.`;
        } 
        else if (neededMarks <= 0) {
            alertBox.classList.add('alert-success');
            alertHTML = `🎉 Excellent work! You have already surpassed the 80 mark threshold. You need <strong>0</strong> more marks.`;
        } 
        else {
            if (neededMarks <= 40) {
                alertHTML = `You need <strong>${neededMarks.toFixed(2)}</strong> more marks to hit 80.<br><br>Target <strong>${neededMarks.toFixed(2)} out of 40</strong> on the Final Exam.`;
            } else {
                alertHTML = `You need <strong>${neededMarks.toFixed(2)}</strong> more marks to hit 80.<br><br>⚠️ <strong>Warning:</strong> A maximum of 40 marks are available on the Final Exam. You must secure missing assignment or quiz marks to reach your goal.`;
            }

            if (quizzesTaken > 0 && quizzesTaken < 3) {
                let potentialBoost = 15 / 3; 
                let newNeededFinal = neededMarks - potentialBoost;
                if (newNeededFinal <= 40) {
                    alertHTML += `<br><br>💡 <em>Tip: Score 15/15 on your next quiz to drop your final exam requirement to <strong>${newNeededFinal.toFixed(2)}</strong>.</em>`;
                }
            }
        }

        alertBox.innerHTML = alertHTML;
        setTimeout(() => alertBox.classList.add('show'), 400);
    }
</script>
